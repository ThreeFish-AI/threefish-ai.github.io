еүҚеҗ‘дј ж’ӯпјҲForward Propagationпјүпјҡе°Ҷиҫ“е…Ҙж•°жҚ®йҖҡиҝҮзҪ‘з»ңиҝӣиЎҢиҝҗз®—пјҢеҫ—еҲ°зҪ‘з»ңзҡ„иҫ“еҮәгҖҒиҫ“еҮәдёҺзӣ®ж ҮеҖјд№Ӣй—ҙзҡ„иҜҜе·®гҖӮ йҖҶдј ж’ӯпјҲBackward Propagationпјүпјҡд»Һиҫ“еҮәеұӮејҖе§ӢпјҢе°ҶиҜҜе·®йҖҶдј ж’ӯеҲ°йҡҗи—ҸеұӮпјҢзӣҙеҲ°иҫ“е…ҘеұӮгҖӮйҖҶдј ж’ӯиҝҮзЁӢеҸҜд»Ҙи®Ўз®—жҜҸдёӘжқғйҮҚзҡ„жўҜеәҰпјҢеҚіиҜҜе·®зӣёеҜ№дәҺжҜҸдёӘжқғйҮҚзҡ„еҒҸеҜјж•°гҖӮ

    """

    """

    def __init__(self):

        self.y = None

    def forward(self, x, y):
        """
        еүҚеҗ‘дј ж’ӯ

        Args:
            x: иҫ“е…Ҙ x
            y: иҫ“е…Ҙ y

        Returns:
            out: иҫ“еҮә
        """

        out = x + y

        """

        dy = dout * 1

        return dx, dy

еҗҢдёҠпјҢжҲ‘д»¬з”Ё MulLayer зұ»е®һзҺ°д№ҳжі•иҝҗз®—зҡ„еүҚеҗ‘дј ж’ӯдёҺйҖҶдј ж’ӯпјҡ

class MulLayer:
    """
    д№ҳжі•иҝҗз®—зҡ„еүҚеҗ‘дј ж’ӯдёҺйҖҶдј ж’ӯ
    """

    def __init__(self):
        self.x = None
        self.y = None

    def forward(self, x, y):
        """
        еүҚеҗ‘дј ж’ӯ

        Args:
            x: иҫ“е…Ҙ x
            y: иҫ“е…Ҙ y

        Returns:
            out: иҫ“еҮә
        """

        self.x = x
        self.y = y
        out = x * y

        return out

    def backward(self, dout):
        """
        йҖҶдј ж’ӯ

        Args:
            dout: дёҠжёёжўҜеәҰ

        Returns:
            dx: x зҡ„жўҜеәҰ
            dy: y зҡ„жўҜеәҰ
        """

        dx = dout * self.y
        dy = dout * self.x

        return dx, dy

SoftmaxWithLoss еұӮвҖӢ

жҲ‘д»¬зҹҘйҒ“зҘһз»ҸзҪ‘з»ңжЁЎеһӢзҡ„и®ӯз»ғиҝҮзЁӢе°ұжҳҜж №жҚ®жҚҹеӨұеҮҪж•°е…ідәҺжқғйҮҚеҸӮж•°зҡ„жўҜеәҰдјҳеҢ–жқғйҮҚеҸӮж•°зҡ„иҝҮзЁӢпјҢе…¶дёӯжұӮи§ЈжҚҹеӨұеҮҪж•°е…ідәҺжқғйҮҚеҸӮж•°зҡ„жўҜеәҰжҳҜиҝҗз®—зҡ„ж ёеҝғгҖӮиҖҢжҚҹеӨұеҮҪж•°еҫҖеҫҖжҳҜзҘһз»ҸзҪ‘з»ңжӯЈеҗ‘дј ж’ӯдёӯзҡ„жңҖеҗҺдёҖдёӘзҺҜиҠӮпјҲи®ӯз»ғиҝҮзЁӢзҡ„жңҖеҗҺдёҖдёӘиҝҮзЁӢжҳҜжҚҹеӨұеҮҪж•°пјҢжҺЁзҗҶиҝҮзЁӢеҲҷдёҖиҲ¬дёҚйңҖиҰҒи®Ўз®—жҚҹеӨұпјүпјҢж №жҚ® BP з®—жі•зҡ„жҖқи·ҜпјҢеңЁйҖҶдј ж’ӯиҝҮзЁӢдёӯпјҢжұӮи§ЈжҚҹеӨұеҮҪж•°зҡ„вҖңеұҖйғЁжўҜеәҰвҖқе°ұжҲҗдәҶжұӮи§ЈжқғйҮҚеҸӮж•°жўҜеәҰзҡ„з¬¬дёҖжӯҘгҖӮ

з”ұдәҺеңЁеӨҡеҲҶзұ»д»»еҠЎдёӯпјҢзҘһз»ҸзҪ‘з»ңжЁЎеһӢз»ҸеёёдҪҝз”Ё Softmax еҮҪж•°жқҘеҜ№жңҖз»Ҳиҫ“еҮәеҒҡеҪ’дёҖеҢ–еӨ„зҗҶпјҢжҲ‘д»¬еңЁе°ҒиЈ…жҚҹеӨұеҮҪж•°ж—¶пјҢйҖҡеёёдјҡе°Ҷ Softmax еҮҪж•°дёҺжҚҹеӨұеҮҪж•°з»“еҗҲеңЁдёҖиө·пјҢиҝҷж ·зҡ„з»“жһ„жҲ‘д»¬з§°д№ӢдёәSoftmaxWithLossеұӮгҖӮ

дёӢйқўжҲ‘д»¬д»ҘдәӨеҸүзҶөиҜҜе·®дёәдҫӢпјҢйҖҡиҝҮе®һзҺ°дёҖдёӘ SoftmaxWithLoss еұӮжқҘжј”зӨә BP з®—жі•еҸҠе…¶вҖңеұҖйғЁжўҜеәҰвҖқзҡ„жұӮи§ЈиҝҮзЁӢгҖӮ

еҒҮе®ҡзҪ‘з»ңзҡ„иҫ“еҮәеұӮжңү n дёӘзҘһз»Ҹе…ғпјҲn дёӘеҲҶзұ»зұ»еҲ«пјүпјҢеҲҷ SoftmaxWithLoss еұӮзҡ„и®Ўз®—иҝҮзЁӢеҰӮеӣҫ 3 жүҖзӨәпјҡ

еӣҫ 3пјҡSoftmaxWithLoss зҡ„и®Ўз®—иҝҮзЁӢ

д»ҺеүҚйқўзҡ„еұӮиҫ“е…Ҙзҡ„жҳҜ (a1,a2,...,an)(a_1, a_2, ..., a_n)(a1вҖӢ,a2вҖӢ,...,anвҖӢ)пјҢSoftmax еұӮиҫ“еҮәзҡ„жҳҜ (y1,y2,...,yn)(y_1, y_2, ..., y_n)(y1вҖӢ,y2вҖӢ,...,ynвҖӢ)пјҢе®һйҷ…з»“жһңеҲҶеҲ«жҳҜ (t1,t2,...,tn)(t_1, t_2, ..., t_n)(t1вҖӢ,t2вҖӢ,...,tnвҖӢ)пјҢCross Entropy Error иҫ“еҮәзҡ„жҚҹеӨұжҳҜ LгҖӮ

жӯЈеҗ‘дј ж’ӯвҖӢ

Softmax еҫҖеҫҖдҪңдёәзҪ‘з»ңиҫ“еҮәеұӮзҡ„жҝҖжҙ»еҮҪж•°пјҢеҜ№зҪ‘з»ңзҡ„иҫ“еҮәеҒҡжңҖеҗҺзҡ„еҪ’дёҖеҢ–еӨ„зҗҶпјӣиҖҢеңЁжЁЎеһӢи®ӯз»ғж—¶пјҢSoftmax зҡ„иҫ“еҮәдёҺе®һйҷ…з»“жһңдҪңдёәжҚҹеӨұеҮҪж•°зҡ„иҫ“е…ҘпјҢеҸҜд»Ҙи®Ўз®—еҮәжЁЎеһӢи®ӯз»ғжүҖйңҖзҡ„жҚҹеӨұеҖјгҖӮеҸҜи§Ғ SoftmaxWithLoss еұӮе®һйҷ…е°ұжҳҜз»ҸиҝҮдәҶ Softmax и®Ўз®—е’Ң Loss Function и®Ўз®—дёӨдёӘиҝҮзЁӢгҖӮ

е…¶дёӯ Softmax и®Ўз®—иҝҮзЁӢеҰӮејҸ 2пјҡ

class SoftmaxWithLoss:
    def __init__(self):
        self.loss = None
        self.y = None  # softmaxзҡ„иҫ“еҮә
        self.t = None  # зӣ‘зқЈж•°жҚ®

    def forward(self, x, t):
        """
        еүҚеҗ‘дј ж’ӯ

        Args:
            x: иҫ“е…Ҙж•°жҚ®
            t: зӣ‘зқЈж•°жҚ®

        Returns:
            float: жҚҹеӨұ
        """

        self.t = t
        self.y = softmax(x)
        self.loss = cross_entropy_error(self.y, self.t)

        return self.loss

    def backward(self, dout=1):
        """
        йҖҶдј ж’ӯ

        Args:
            dout: дёҠжёёжўҜеәҰ

        Returns:
            np.ndarray: жҚҹеӨұе…ідәҺиҫ“е…Ҙ x зҡ„жўҜеәҰ
        """

        batch_size = self.t.shape[0]
        if self.t.size == self.y.size:  # зӣ‘зқЈж•°жҚ®жҳҜone-hot-vectorзҡ„жғ…еҶө
            dx = (self.y - self.t) / batch_size
        else:
            dx = self.y.copy()
            dx[np.arange(batch_size), self.t] -= 1
            dx = dx / batch_size

        return dx

import numpy as np

def cross_entropy_error(y, t):
    """
    дәӨеҸүзҶөиҜҜе·®еҮҪж•°

    Args:
        y: зҘһз»ҸзҪ‘з»ңзҡ„иҫ“еҮә
        t: зӣ‘зқЈж•°жҚ®

    Returns:
        float: дәӨеҸүзҶөиҜҜе·®
    """

    # зӣ‘зқЈж•°жҚ®жҳҜ one-hot-vector зҡ„жғ…еҶөдёӢпјҢиҪ¬жҚўдёәжӯЈзЎ®и§Јж Үзӯҫзҡ„зҙўеј•
    if t.size == y.size:
        t = t.argmax(axis=1)

    batch_size = y.shape[0]
    return -np.sum(np.log(y[np.arange(batch_size), t] + 1e-7)) / batch_size

def softmax(x):
   """еҪ’дёҖеҢ–жҢҮж•°еҮҪж•°"""
   if x.ndim == 2:
       x = x.T
       x = x - np.max(x, axis=0)
       y = np.exp(x) / np.sum(np.exp(x), axis=0)
       return y.T

   x = x - np.max(x)  # жәўеҮәеҜ№зӯ–
   return np.exp(x) / np.sum(np.exp(x))
