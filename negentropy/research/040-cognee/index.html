  2. и®ӨзҹҘеҢ– (Cognify)пјҡ

    • зӨәдҫӢе…ізі»пјҡAUTHORED_BY, CITES, USES_METHOD.

  3. иҜӯд№үеўһејә (Memify)пјҡ

    • (еҸҜйҖү) й’ҲеҜ№зү№е®ҡйўҶеҹҹйҖ»иҫ‘пјҢиҝӣдёҖжӯҘдё°еҜҢеӣҫи°ұгҖӮдҫӢеҰӮпјҢе»әз«Ӣ "Methodology Hierarchy"пјҲиҝҷдёҖж–№жі•и®әеұһдәҺе“ӘдёҖзұ»пјүгҖӮ

  4. жЈҖзҙўдёҺеўһејә (Retrieval)пјҡ

    • еңЁ Heartfelt Agent иҝӣиЎҢж·ұеәҰеҲҶжһҗж—¶пјҢдёҚд»…жЈҖзҙўзӣёдјјж–Үжң¬пјҢиҝҳжҹҘиҜўеӣҫи°ұпјҡ

from agents.core.memory import CogneeMemory
from agents.heartfelt.agent import HeartfeltAgent

class CogneeMemory:
    def __init__(self):
        cognee.config.set_llm_provider("anthropic")
        cognee.config.set_llm_model("claude-3-5-sonnet-20241022")

    async def add_paper(self, content: str, metadata: dict):
        """ж·»еҠ и®әж–ҮеҲ°и®°еҝҶ"""
        await cognee.add(content, dataset_name="papers")
        await cognee.cognify()

    async def search(self, query: str, search_type: str = "INSIGHTS"):
        """жЈҖзҙўзӣёе…ізҹҘиҜҶ"""
        type_map = {
            "INSIGHTS": SearchType.INSIGHTS,
            "GRAPH": SearchType.GRAPH_TRAVERSAL,
            "SEMANTIC": SearchType.CHUNKS
        }
        return await cognee.search(
            type_map.get(search_type, SearchType.INSIGHTS),
            query_text=query
        )

# Agent дҪҝз”ЁзӨәдҫӢ
async def agent_with_memory():
    memory = CogneeMemory()
    agent = HeartfeltAgent(memory=memory)

    # Agent жҹҘиҜўж—¶иҮӘеҠЁдҪҝз”Ёи®°еҝҶ
    response = await agent.analyze(
        "жҜ”иҫғ ReAct е’Ң Chain-of-Thought еңЁиЎЁж јжҺЁзҗҶд»»еҠЎдёҠзҡ„иЎЁзҺ°"
    )
    print(response)
